SELECT mnev, AVG(nepesseg) AS Átlag
	FROM varos INNER JOIN megye ON varos.megyeid = megye.id
    WHERE vnev <> 'Budapest'
    GROUP BY mnev
    ORDER BY Átlag ASC;

3.)
SELECT mnev, CONCAT(AVG(nepesseg), ' fő') AS Átlag
	FROM varos INNER JOIN megye ON varos.megyeid = megye.id
    WHERE vnev <> 'Budapest'
    GROUP BY mnev
    ORDER BY Átlag ASC;

4.)
SELECT mnev, COUNT(varos.id) AS Településszám
	FROM varos INNER JOIN megye ON varos.megyeid = megye.id
    WHERE vnev <> 'Budapest' AND LENGTH(mnev) = 5
    GROUP BY mnev 
    ORDER BY COUNT(varos.id) DESC

5.)
SELECT mnev, CONCAT('-', ROUND(SUM(terulet)), ' km2') AS Terület
	FROM varos INNER JOIN megye ON varos.megyeid = megye.id
    WHERE vnev <> 'Budapest'
    GROUP BY mnev
    ORDER BY ROUND(SUM(terulet)) DESC
    LIMIT 3;

6.)
SELECT CONCAT(LENGTH(varos.vnev), ' kar.') AS Névhossz, COUNT(varos.id) as Városszám
	FROM varos
    WHERE LENGTH(varos.vnev) > 9 OR LENGTH(varos.vnev) < 100
    GROUP BY LENGTH(varos.vnev)
    HAVING Városszám > 20
    ORDER BY Névhossz ASC

SELECT LEFT(vnev, 1) AS Kezdőbetű, COUNT(varos.id) AS DB
	FROM varos
    WHERE vnev <> 'Budapest'
    GROUP BY Kezdőbetű
    ORDER BY Kezdőbetű ASC;
